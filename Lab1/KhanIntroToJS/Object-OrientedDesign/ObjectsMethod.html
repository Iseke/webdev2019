var SmileFace = function(centerX,centerY){ this.centerX = centerX; this.centerY = centerY; }; SmileFace.prototype.draw = function() { fill(255, 234, 0); ellipse(this.centerX, this.centerY, 150, 150); fill(0, 0, 0); ellipse(this.centerX-30, this.centerY-30, 20, 20); ellipse(this.centerX+30, this.centerY-30, 20, 20); noFill(); strokeWeight(3); arc(this.centerX, this.centerY+10, 64, 40, 0, 180); }; SmileFace.prototype.speak = function(sp){ text(sp,this.centerX,this.centerY+100); }; var Face = new SmileFace(200,100); Face.draw(); Face.speak("My name is Smiley"); var Eblet = new SmileFace(200,280); Eblet.draw(); Eblet.speak("I'm Davron Zombarev");
